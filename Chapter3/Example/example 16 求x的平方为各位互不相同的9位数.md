### 例16
求x，使x^2为一个各位数字互不相同的9位数。
### 问题分析
由于x^2为9位数，估算x应该在10000 ~ 32000之间。
### 算法设计
```c
#include<stdio.h>
int main()
{
    int i, x, n, t, s, a[10];
    for (x = 10000;x < 32000;x++)
    {
        for (i = 0;i < 10;i++)      //用数组a来存储x^2各个位数出现的次数
            a[i] = 0;               //元素等于0表示该位数未出现
        s = 0;
        n = x * x;
        while (n > 0)
        {
            t = n % 10;
            if (a[t] == 1)          //用位数作为数组下标，若元素等于1，则表示该数已出现过一次，不能重复出现
                break;
            else
            {
                a[t] = 1;           //若该数未出现，则赋值为1 ，表示已出现
                s++;                //用s作为位数出现的计数器，当s等于9时，说明各位数均已出现过一次
            }
            n /= 10;
        }
        if (s == 9)
            printf("%d * %d = %d\n", x, x, x * x);
    }
    printf("\n");
    return 0;
}
```
